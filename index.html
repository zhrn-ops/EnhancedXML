XSD Schema Analysis

JATS XML Schema (Input)

Key Elements:

<article> - Root element with article-type attribute
<front> - Contains metadata and journal information
<article-meta> - Article metadata container
<abstract> - Article abstract with paragraphs
<kwd-group> - Keywords group
<counts> - Page count information

Validation Rules:

  • Mixed content allowed for text elements
  • Optional attributes with use="optional"
  • Unbounded occurrence for repeatable elements
  • Type constraints (xs:string, xs:byte, xs:short)

Namespaces:

xmlns:xs="http://www.w3.org/2001/XMLSchema"
xmlns:xlink="http://www.w3.org/1999/xlink"

Elsevier XML Schema (Output)

Key Elements:

<doi> - Digital Object Identifier
<given-name> - Author first name
<surname> - Author last name
<initials> - Author initials
<e-address> - Email address with type attribute
<source-text> - Source text content
<para> - Paragraph content

Target Namespace:

targetNamespace="http://www.elsevier.com/xml/ani/common"

Element Characteristics:

  • Simple string types for most elements
  • Complex type for e-address with type attribute
  • Extension base for attribute combinations

Enhanced XSD Validation Implementation 

class EnhancedXMLMapper:
    def __init__(self, jats_xsd_path, elsevier_xsd_path):
        self.jats_schema = etree.XMLSchema(etree.parse(jats_xsd_path))
        self.elsevier_schema = etree.XMLSchema(etree.parse(elsevier_xsd_path))
        self.validation_errors = []
        
    def validate_input(self, xml_doc):
        # Validate JATS XML against XSD schema
        if not self.jats_schema.validate(xml_doc):
            for error in self.jats_schema.error_log:
                self.validation_errors.append({
                    'type': 'input_validation',
                    'line': error.line,
                    'message': error.message
                })
            return False
        return True
        
    def validate_output(self, xml_doc):
        # Validate Elsevier XML against XSD schema
        if not self.elsevier_schema.validate(xml_doc):
            for error in self.elsevier_schema.error_log:
                self.validation_errors.append({
                    'type': 'output_validation',
                    'line': error.line,
                    'message': error.message
                })
            return False
        return True

#!/usr/bin/env python3
import unittest
import lxml.etree as etree
from pathlib import Path

class JATSToElsevierTest(unittest.TestCase):
    
    def setUp(self):
        # Load XSD schemas
        self.jats_schema = etree.XMLSchema(
            etree.parse('schemas/jats.xsd')
        )
        self.elsevier_schema = etree.XMLSchema(
            etree.parse('schemas/elsevier.xsd')
        )
        self.mapper = JATSToElsevierMapper()
    
    def test_valid_jats_transformation(self):
        # Test with valid JATS XML
        test_file = Path('test_data/valid_jats.xml')
        jats_doc = etree.parse(test_file)
        
        # Validate input
        self.assertTrue(
            self.jats_schema.validate(jats_doc),
            'Input JATS XML should be valid'
        )
        
        # Transform
        elsevier_doc = self.mapper.transform(jats_doc)
        
        # Validate output
        self.assertTrue(
            self.elsevier_schema.validate(elsevier_doc),
            'Output Elsevier XML should be valid'
        )
        
        # Check content preservation
        self.assert_content_preserved(jats_doc, elsevier_doc)
    
    def test_invalid_input_handling(self):
        # Test error handling for invalid input
        invalid_xml = '<invalid>broken xml</malformed>'
        
        with self.assertRaises(ValidationError):
            self.mapper.transform_from_string(invalid_xml)
    
    def assert_content_preserved(self, jats_doc, elsevier_doc):
        # Verify key content is preserved during transformation
        jats_title = jats_doc.findtext('.//article-title')
        elsevier_title = elsevier_doc.findtext('.//title')
        self.assertEqual(jats_title, elsevier_title)

if __name__ == '__main__':
    unittest.main()
